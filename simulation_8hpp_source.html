163 template <region_compatible R> auto simulate(const simulation_opts_t& opts = {}) -> void
164 {
165  std::mt19937 rnd(opts.seed ? *opts.seed : std::random_device{}());
166 
167  agents_private_status_fn agents;
168  constexpr agents_private_status_accessor accessor;
169  std::vector<id_t> keep_active;
170 
171  uint_t t0 = 0;
172 
173  std::deque<std::unordered_map<permit<R>, permit_private_status_t>> data;
174 
175  permit_private_status_t ool = {permit_private_status::out_of_limits{}, {}};
176  auto book = [&t0, &data, &ool, &opts](region_view loc, uint_t t) mutable -> permit_private_status_t& {
177  if (t < t0) // XXX agents can check the state at t0, however they should be prohibited to bid for.
178  return ool;
179  if (opts.time_window && t > t0 + 1 + *opts.time_window)
180  return ool;
181  while (t - t0 >= data.size())
182  data.emplace_back();
183  return data[t - t0][{loc.downcast<R>(), t}];
184  };
185 
186  const auto safe_book = [&book](region_view loc, uint_t t) -> permit_private_status_t { return book(loc, t); };
187 
188  auto public_access = [&book](auto id) {
189  return [id = id, &book](region_view s, uint_t t) -> permit_public_status_t {
190  using namespace permit_private_status;
191  using namespace permit_public_status;
192  const auto& pstatus = book(s, t);
193  using history_t = std::add_const_t<decltype(pstatus.history)>&;
194  return std::visit(cool::compose{[](out_of_limits) -> permit_public_status_t { return unavailable{}; },
195  [&](in_use status) -> permit_public_status_t {
196  return status.owner == id ? permit_public_status_t{owned{}} : unavailable{};
197  },
198  [&](on_sale status) -> permit_public_status_t {
199  return status.owner == id
200  ? permit_public_status_t{unavailable{}}
201  : available{status.min_value, [&]() -> history_t { return pstatus.history; }};
202  }},
203  pstatus.current);
204  };
205  };
206 
207  const auto stop = [&] {
208  using namespace stop_criterion;
209  return std::visit(cool::compose{
210  [&](no_agents_t) { return agents.active_count() == 0; },
211  [&](time_threshold_t th) { return t0 > th.t; },
212  },
213  opts.stop_criterion);
214  };
215 
216  do {
217  if (opts.status_callback)
218  opts.status_callback(t0, std::as_const(agents), safe_book);
219 
220  // Generate new agents
221  if (opts.factory) {
222  auto new_agents = opts.factory(t0, rnd());
223  for (auto& agent : new_agents)
224  accessor.insert(agents, std::move(agent));
225  }
226 
227  {
228  // Bid phase
229  std::vector<permit<R>> bids;
230  for (const auto id : accessor.active(agents)) {
231  auto bid = [&](region_view s, uint_t t, value_t v) -> bool {
232  if (t < t0)
233  return false;
234  using namespace permit_private_status;
235  const auto visitor = cool::compose{[](out_of_limits) { return false; }, [](in_use) { return false; },
236  [&](on_sale& status) {
237  if (v > status.min_value && v > status.highest_bid) {
238  if (status.highest_bidder == no_owner)
239  bids.emplace_back(s.downcast<R>(), t);
240  status.highest_bidder = id;
241  status.highest_bid = v;
242  }
243  return true;
244  }};
245  return std::visit(visitor, book(s, t).current);
246  };
247 
248  accessor.at(agents, id).bid_phase(t0, std::move(bid), public_access(id), rnd());
249  }
250 
251  // Trading
252  if (bids.size() > 0) {
253  const auto first_active = accessor.active(agents).front();
254  for (const auto& [s, t] : bids) {
255  const auto status = std::get<permit_private_status::on_sale>(book(s, t).current);
256  if (opts.trade_callback)
257  opts.trade_callback({t0, status.owner, status.highest_bidder, s, t, status.highest_bid});
258 
259  accessor.at(agents, status.highest_bidder).on_bought(s, t, status.highest_bid);
260  if (status.owner != no_owner && status.owner >= first_active)
261  accessor.at(agents, status.owner).on_sold(s, t, status.highest_bid);
262 
263  auto& pstatus = book(s, t);
264  pstatus.current = permit_private_status::in_use{status.highest_bidder};
265  pstatus.history.push_back({status.min_value, status.highest_bid});
266  }
267  }
268  }
269 
270  // Ask phase
271  {
272  std::vector<std::tuple<R, uint_t, uint_t, value_t>> asks;
273  for (const auto id : accessor.active(agents)) {
274  auto ask = [&](region_view s, uint_t t, value_t v) -> bool {
275  if (t < t0)
276  return false;
277  using namespace permit_private_status;
278  const auto visitor = cool::compose{[](out_of_limits) { return false; },
279  [&](on_sale status) {
280  if (status.owner != id)
281  return false;
282  asks.emplace_back(s.downcast<R>(), t, id, v);
283  return true;
284  },
285  [&](in_use& status) {
286  if (status.owner != id)
287  return false;
288  asks.emplace_back(s.downcast<R>(), t, id, v);
289  return true;
290  }};
291  return std::visit(visitor, book(s, t).current);
292  };
293 
294  accessor.at(agents, id).ask_phase(t0, std::move(ask), public_access(id), rnd());
295  }
296 
297  for (const auto& [s, t, id, v] : asks)
298  book(s, t).current = permit_private_status::on_sale{.owner = id, .min_value = v};
299  }
300 
301  // Stop condition
302  keep_active.clear();
303  keep_active.reserve(agents.active_count());
304  for (const auto id : accessor.active(agents))
305  if (!accessor.at(agents, id).stop(t0, rnd()))
306  keep_active.push_back(id);
307  accessor.update_active(agents, std::move(keep_active));
308 
309  if (data.size() > 0)
310  data.pop_front();
311  ++t0;
312  } while (!stop());
313 }
